import { Component, OnInit, Input, TemplateRef } from '@angular/core';
import {
  SticosWidget,
  SticosWidgetMeta,
} from '../../decorators/sticos-widget.decorator';

import { WidgetService } from '../../services/widget.service';
import { IVacationBankSettings } from './models/vacation-bank-input';
import { FormGroup, FormBuilder } from '@angular/forms';

@Component({
  selector: 'sw-vacation-bank-widget',
  templateUrl: './vacation-bank-widget.component.html',
  styleUrls: ['./vacation-bank-widget.component.scss'],
})
@SticosWidget({
  widgetName: 'VacationBankWidget',
  headerName: 'Vacation bank',
  options: {
    defaultHeight: 3,
    defaultWidth: 4,
    minHeight: 3,
    minWidth: 4,
  },
})
export class VacationBankWidgetComponent implements OnInit {
  popupVisible = false;
  vacationBankForm: FormGroup;
  days = 0;
  @Input()
  mod: string;
  @Input()
  settings: IVacationBankSettings;
  @Input()
  widgetId: string;
  @Input()
  loadingTemplate: TemplateRef<any>;

  @SticosWidgetMeta(VacationBankWidgetComponent)
  sticosWidgetOptions: any;

  constructor(private widgetService: WidgetService, private fb: FormBuilder) {}

  ngOnInit() {
    this.createForm();

    if (!this.mod) {
      this.mod = 'default';
    }
    if (!this.settings) {
      this.settings = {
        days: this.days,
      };
    }
  }

  createForm() {
    this.vacationBankForm = this.fb.group({
      days: 0,
    });
  }

  setVacationDays() {
    this.days = this.vacationBankForm.value['days'];
    this.settings = {
      days: this.days,
    };
    this.widgetService.settingsChanged({
      settings: this.settings,
      widgetId: this.widgetId,
    });
    this.popupVisible = false;
  }

  onDeleteWidget() {
    this.widgetService.deleteWidget(this.widgetId);
  }

  onShowSettings() {
    this.vacationBankForm.value['days'] = this.days;
    this.popupVisible = true;
  }
}
