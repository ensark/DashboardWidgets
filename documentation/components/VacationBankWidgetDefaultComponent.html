import { Component, OnInit, Input } from '@angular/core';
import { IVacationBankSettings } from '../models/vacation-bank-input';

@Component({
  selector: 'sw-vacation-bank-widget-default',
  templateUrl: './vacation-bank-widget-default.component.html',
  styleUrls: ['./vacation-bank-widget-default.component.scss'],
})
export class VacationBankWidgetDefaultComponent implements OnInit {
  loadingData: boolean;
  maxValue = 25;
  intervalId: number;

  @Input()
  settings: IVacationBankSettings;
  @Input()
  widgetId: string;

  constructor() {}

  ngOnInit() {}

  format(value) {
    return Math.round(value / 4) * 100;
  }
}
